mariadb安装

卸载数据库： yum -y remove mari
删除数据库文件：rm -rf /var/lib/mysql/
安装: yum -y install mariadb mariadb-server
启动: systemctl start mariadb
查看状态: systemctl status mysqld
系统开启自启动:systemctl enable mariadb

自定义配置：mysql_secure_installation，（比如设置密码 默认密码为空）

添加远程连接权限账号: CREATE USER ‘test’@’%’ IDENTIFIED BY ‘123456’;
GRANT ALL ON . TO ‘test’@’%’;

SELECT
da.name as 城市
,CASE ac.type WHEN 2 THEN ‘二房东(分散式)’ WHEN 3 THEN ‘公寓(集中式)’ END 机构类型
,ac.id as 机构编号
,ac.name as 机构名称
,c.id as 租约编号
,u.name as 租客姓名
,u.mobile as 租客手机
,c.term 租期
,ddt.name as 导客费收取类型
,o.id as 签约人编号
,o.nickname as 签约人
,c.month_rent as 租约月租金
,CASE WHEN c.status = 2 THEN ‘未付款’ WHEN c.status=3 THEN ‘已付款’ END 账单付款状态
,CASE c.flag WHEN 0 THEN ‘正常’ WHEN 1 THEN ‘删除’ WHEN 2 THEN ‘过期’ WHEN 7 THEN ‘已关闭’ WHEN 8 THEN ‘已终止’ WHEN 9 THEN ‘已作废’ END 租约状态
,CASE WHEN c.is_monthly_pay = 1 THEN ‘付一押一’ ELSE dwr.name END as 租客缴租方式
, wb.name as 板块
,from_unixtime(c.first_trade_time) as 租约首次付款时间
,date_sub(from_unixtime(c.first_trade_time,’%Y-%m-%d’),INTERVAL WEEKDAY(from_unixtime(c.first_tradetime,’%Y-%m-%d’)) DAY) as 租约首次付款日期周
,from_unixtime(c.first_tradetime,’%Y-%m’) as 租约首次付款日期年月
, 据点
, B.period_number 租客账单期数
, FROM_UNIXTIME(B.start_date) 账单起始日
, WeekMysql(B.startdate) 账单起始日周
, dbo.f_FROM_UNIXTIME(B.end_date) 账单结束日
, dbo.f_FROM_UNIXTIME(B.trade_time) 付款日期
, dbo.f_WeekMysql(B.tradetime) 付款日期周
, dbo.f_MonthMysql(B.tradetime) 付款日期年月
, dbo.f_DayMysql(B.pay_date) 账单约定付款日
, 房东PC版
, 提前缴租天数
, 导客费
, CASE WHEN B.flag = 0 THEN ‘正常’ WHEN B.flag = 1 THEN ‘删除’ END 账单是否正常
, CASE WHEN B.status = ‘0’ THEN ‘未支付’ WHEN B.status = ‘1’ THEN ‘已支付’ WHEN B.status = ‘7’ THEN ‘已作废’ WHEN B.status = ‘9’ THEN ‘已关闭’ END 账单状态
, B.bill_all_id 账单编号
, cclhd.是否月付
, cclhd.营运服务
, CASE WHEN ba.is_line_pay = 1 THEN ‘是’ ELSE ‘否’ END 是否虚拟核销
, cclhd.是否代管
, verificationUser.Name_full 核销人
, cclhd.租约类型
FROM bill b
LEFT JOIN bill_all ba on ba.id=b.bill_all_id
left join contract c on c.id=b.contract_id
LEFT JOIN dd_way_rent dwr ON c.way_rent = dwr.id
left join user u on u.id=c.user_id
left join operator o on o.user_id=c.sign_user_id
left join work_group wg on wg.id=o.group_id
left join work_group ws on ws.id=wg.fid
left join work_group wb on wb.id=ws.fid
left join dd_area as da on da.id=c.city_id
left join agency_compnay ac on ac.id=c.land_company_id
left join user u on u = ba.verification_user_id
